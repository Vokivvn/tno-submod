TNO_Russia_load_regional_tree = {
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Andrei Zhdanov"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_zhdanov_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Svetlana Bukharina"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_bukharina_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_government = communist
		}
		load_focus_tree = NIK_suslov_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Svetlana Stalina"
				ruling_only = yes
			}
			has_government = despotism
		}
		load_focus_tree = NIK_stalina_despot_regional
	}
	if = {
		limit = {
			original_tag = NIK
			OR = {
				has_government = social_democrat
				has_government = social_liberal
				has_government = authoritarian_democrat
			}
		}
		load_focus_tree = NIK_second_election_tree
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Ivan Serov"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_serov_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Igor Shafarevich"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_shafarevich_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Sergey Taboritsky"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_taboritsky_regional
	}
	if = {
		limit = {
			original_tag = NIK
			has_country_leader = {
				name = "Lev Gumilyov"
				ruling_only = yes
			}
		}
		load_focus_tree = NIK_gumilyov_regional
	}
}

RUS_decrease_administrative_strain = {
	if = {
		limit = {
			has_idea = RUS_administrative_overload_1
		}
		hidden_effect = {
			swap_ideas = {
				remove_idea = RUS_administrative_overload_1
				add_idea = RUS_administrative_overload_2
			}
		}
		custom_effect_tooltip = RUS_strain_reduce
	}
	else_if = {
		limit = {
			has_idea = RUS_administrative_overload_2
		}
		hidden_effect = {
			swap_ideas = {
				remove_idea = RUS_administrative_overload_2
				add_idea = RUS_administrative_overload_3
			}
		}
		custom_effect_tooltip = RUS_strain_reduce
	}
	else_if = {
		limit = {
			has_idea = RUS_administrative_overload_3
		}
		hidden_effect = {
			swap_ideas = {
				remove_idea = RUS_administrative_overload_3
				add_idea = RUS_administrative_overload_4
			}
		}
		custom_effect_tooltip = RUS_strain_reduce
	}
	else_if = {
		limit = {
			has_idea = RUS_administrative_overload_4
		}
		hidden_effect = {
			swap_ideas = {
				remove_idea = RUS_administrative_overload_4
				add_idea = RUS_administrative_overload_5
			}
		}
		custom_effect_tooltip = RUS_strain_reduce
	}
	else_if = {
		limit = {
			has_idea = RUS_administrative_overload_5
		}
		hidden_effect = {
			remove_ideas = RUS_administrative_overload_5
		}
		custom_effect_tooltip = RUS_strain_remove
	}
	else_if = {
		limit = {
			NOT = {
				OR = {
					has_idea = RUS_administrative_overload_5
					has_idea = RUS_administrative_overload_4
					has_idea = RUS_administrative_overload_3
					has_idea = RUS_administrative_overload_2
					has_idea = RUS_administrative_overload_1
				}
			}
		}
		add_stability = 0.05
	}
}

				limit = {

					limit = {
						NOT = {
							OR = {
								tag = ZZZ
								tag = USA
								tag = DSR
								is_russian_nation = yes
							}
						}
						has_government = social_democrat
					}
					country_event = {
						id = socintern.1
						days = 3
						random = 36
					}
				}
				every_other_country = {
					limit = {
						OR = {
							AND = {
								is_russian_nation = yes
								has_government = social_democrat
							}
							has_government = communist
							has_government = socialist
						}
						NOT = {
							OR = {
								tag = ZZZ
