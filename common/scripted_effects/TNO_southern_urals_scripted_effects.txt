owb_throw_victor = {
	#Lysenko
	if = {
		limit = { ROOT = { original_tag = BAS NOT = { has_global_flag = BAS_NKVD_coup } } }
		#Dirlewanger
		if = { limit = { FROM = { original_tag = ORS } }
			ORS = { country_event = { id = ORS.2000 } }
		}
		#Orenburg
		if = { limit = { FROM = { original_tag = ORE } }
			ORE = {country_event = { id = ORE.601 } }
		}
		#Ural League
		if = { limit = { FROM = { original_tag = URL } }
			URL = { country_event = { id = URL.200 } }
		}
	}

	#NKVD
	if = {
		limit = { ROOT = { original_tag = BAS has_global_flag = BAS_NKVD_coup } }
		#Dirlewanger
		if = { limit = { FROM = { original_tag = ORS } }
			ORS = { country_event = { id = ORS.2003 } }
		}
		#Orenburg
		if = { limit = { FROM = { original_tag = ORE } }
			ORE = { country_event = { id = ORE.602 } }
		}
		#Ural League
		if = { limit = { FROM = { original_tag = URL } }
			URL = { country_event = { id = URL.201 } }
		}
	}

	#Dirlewanger
	if = {
		limit = { ROOT = { original_tag = ORS } }

		if = { limit = { FROM = { original_tag = BAS } }

			if = { limit = { has_global_flag = BAS_NKVD_coup } BAS = { country_event = { id = BAS_NKVD.2000 } } }

			else = { BAS = { country_event = { id = BAS.300 } } }
		}

		if = { limit = { FROM = { original_tag = ORE } }
			ORE = { country_event = { id = ORE.600 } }
		}

		if = { limit = { FROM = { original_tag = URL } }
			URL = { country_event = { id = URL.202 } }
		}
	}

	#Orenburg
	if = {
		limit = { ROOT = { original_tag = ORE } }
		#Magnitogorsk
		if = { limit = { FROM = { original_tag = BAS } }
			#NKVD
			if = { limit = { has_global_flag = BAS_NKVD_coup } BAS = { country_event = { id = BAS_NKVD.2002 } } }
			#Lysenko
			else = { BAS = { country_event = { id = BAS.302 } } }
		}
		#Dirlewanger
		if = { limit = { FROM = { original_tag = ORS } }
			ORS = { country_event = { id = ORS.2002 } }
		}
	}

	#Ural League
	if = {
		limit = { ROOT = { original_tag = URL } }
		#Magnitogorsk
		if = { limit = { FROM = { original_tag = BAS } }
			#NKVD
			if = { limit = { has_global_flag = BAS_NKVD_coup } BAS = { country_event = { id = BAS_NKVD.2001 } } }
			#Lysenko
			else = { BAS = { country_event = { id = BAS.301 } } }
		}
		#Dirlewanger
		if = { limit = { FROM = { original_tag = ORS } }
			ORS = { country_event = { id = ORS.2001 } }
		}
	}
}
