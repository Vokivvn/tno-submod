NIK_generate_coup = {
	clr_country_flag = NIK_COUP_TICKER_RUN
	set_variable = { NIKCoupTicker = 1 }
	NIK_shuffle_couped_politician = yes
	if = {
		limit = {
			OR = {
				has_government = social_liberal
				has_government = social_democrat
				has_government = authoritarian_democrat
				has_government = social_conservative
			}
		}
		if = {
			limit = {
				check_variable = { NIKLeftPower > NIKRightPower }
			}
			NIK_get_most_likely_coupist_left = yes
		}
		else_if = {
			limit = {
				check_variable = { NIKRightPower > NIKLeftPower }
			}
			NIK_get_most_likely_coupist_right = yes
		}
		else = {
			NIK_get_most_likely_coupist_left = yes
		}
	}
	if = {
		limit = {
			OR = {
				has_government = socialist
				has_government = communist
			}
		}
		if = {
			limit = {
				check_variable = { NIKCenterPower > NIKRightPower }
			}
			NIK_get_most_likely_coupist_center = yes
		}
		else_if = {
			limit = {
				check_variable = { NIKRightPower > NIKCenterPower }
			}
			NIK_get_most_likely_coupist_right = yes
		}
		else = {
			NIK_get_most_likely_coupist_right = yes
		}
	}
	if = {
		limit = {
			OR = {
				has_government = fascism
				has_government = national_socialism
				has_government = ultranational_socialism
				has_government = despotism
			}
		}
		if = {
			limit = {
				check_variable = { NIKLeftPower > NIKCenterPower }
			}
			NIK_get_most_likely_coupist_left = yes
		}
		else_if = {
			limit = {
				check_variable = { NIKCenterPower > NIKLeftPower }
			}
			NIK_get_most_likely_coupist_center = yes
		}
		else = {
			NIK_get_most_likely_coupist_left = yes
		}
	}
}

NIK_get_most_likely_coupist_right = {
	if = {
		limit = {
			OR = {
				check_variable = { NIKGumilyovInfluence > NIKTaboritskyInfluence }
				has_country_flag = NIK_Taboritsky_coup
			}
			NOT = {
				has_country_flag = NIK_Gumilyov_Dead
				has_country_flag = NIK_Gumilyov_Arrested
				has_country_flag = NIK_Gumilyov_Retired
				has_country_flag = NIK_ultranat_coup
			}
		}
		country_event = { #GUMILYOV COUP
			id = komicoup.8
			days = 3
		}
		#remove_ideas = NIK_gumilyov_shadow
	}
	else_if = {
		limit = {
			check_variable = { NIKSerovInfluence > NIKTaboritskyInfluence }
			NOT = {
				has_country_flag = NIK_Serov_dead
				has_country_flag = NIK_Serov_Arrested
				has_country_flag = NIK_Serov_Retired
				has_country_flag = NIK_fascist_coup
			}
		}
		country_event = {  #SEROV COUP
			id = komicoup.4
			days = 3
		}
		#remove_ideas = NIK_gumilyov_shadow
	}
	else_if = {
		limit = {
			OR = {
				check_variable = {
					var = NIKTaboritskyInfluence
					value = NIKGumilyovInfluence
					compare = greater_than_or_equals
				}
				has_country_flag = NIK_ultranat_coup
			}
			NOT = {
				has_country_flag = NIK_Taboritsky_Dead
				has_country_flag = NIK_Taboritsky_Arrested
				has_country_flag = NIK_Taboritsky_Retired
				has_country_flag = NIK_Taboritsky_coup
			}
		}
		country_event = { #TABORITSKY COUP
			id = komicoup.5
			days = 3
		}
		#remove_ideas = NIK_gumilyov_shadow
	}
	else = {
		set_country_flag = NIK_loop_protection_right
		if = {
			limit = {
				OR = {
					has_government = social_conservative
					has_government = social_liberal
					has_government = social_democrat
					has_government = authoritarian_democrat
				}
				NOT = { has_country_flag = NIK_loop_protection_left }
			}
			country_event = komicoup.92
		}
		else_if = {
			limit = {
				OR = {
					has_government = communist
					has_government = socialist
				}
				NOT = { has_country_flag = NIK_loop_protection_center }
			}
			country_event = komicoup.93
		} else = {
			country_event = {
				id = komicoup.20
			}
		}
	}
}
