MZB_REICHSTAAT_compute_devastation_effects = {
	set_variable = { MZB_REICHSTAAT_devastation_resistance_damage = MZB_REICHSTAAT_devastation_score }
	divide_variable = { MZB_REICHSTAAT_devastation_resistance_damage = 250 }
	subtract_from_variable = { MZB_REICHSTAAT_devastation_resistance_damage = 1 }
	multiply_variable = { MZB_REICHSTAAT_devastation_resistance_damage = -1 }
	multiply_variable = { MZB_REICHSTAAT_devastation_resistance_damage = 0.5 }

	if = {
		limit = {
			check_variable = { MZB_REICHSTAAT_devastation_score < 500 }
		}
		set_variable = { MZB_REICHSTAAT_devastation_pop = MZB_REICHSTAAT_devastation_score }
		subtract_from_variable = { MZB_REICHSTAAT_devastation_pop = 100 }
		divide_variable = { MZB_REICHSTAAT_devastation_pop = 400 }
		multiply_variable = { MZB_REICHSTAAT_devastation_pop = -0.75 }
	}

	else = {
		set_variable = { MZB_REICHSTAAT_devastation_pop = -0.75 }
	}

	if = {
		limit = {
			check_variable = {
				var = MZB_REICHSTAAT_devastation_score
				value = 100
				compare = less_than
			}
			NOT = { 
				has_dynamic_modifier = {
				    modifier = MZB_REICHSTAAT_devastation_stage_1
				    scope = MZB
				} 
			}
		}
		MZB = {
			add_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_1 }
			remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_2 }
			remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_3 }
		}
	}

	else = {
		if = {
			limit = {
				check_variable = {
					var = MZB_REICHSTAAT_devastation_score
					value = 250
					compare = less_than
				}
				check_variable = {
					var = MZB_REICHSTAAT_devastation_score
					value = 100
					compare = greater_than_or_equals
				}
				NOT = { 
					has_dynamic_modifier = {
					    modifier = MZB_REICHSTAAT_devastation_stage_2
					    scope = MZB
					} 
				}
			}
			MZB = {
				remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_1 }
				add_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_2 }
				remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_3 }
			}
		}

		else = {
			if = {
				limit = {
					NOT = { 
						has_dynamic_modifier = {
						    modifier = MZB_REICHSTAAT_devastation_stage_3
						    scope = MZB
						} 
					}
				}
				MZB = {
					remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_1 }
					remove_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_2 }
					add_dynamic_modifier = { modifier = MZB_REICHSTAAT_devastation_stage_3 }
				}
			}
		}
	}
}

		log = "[GetDateText]: [?MZB_REICHSTAAT_cabal_power]"
	}
}
