	ENG_DEC_ENGGOOGLEMAPS_HMMLR_TOGGLE_DEBUG = {
		icon = GFX_decision_generic_prepare_civil_war
		visible = {
			is_debug = yes			
		}
		available = {
			is_debug = yes
		}
		complete_effect = {
			if = {
				limit = {
					has_country_flag = ENG_Player_HMMLR
				}
			clr_country_flag = ENG_Player_HMMLR
			set_country_flag = ENG_Player_Collab
			}
			else_if = {
				limit = {
					has_country_flag = ENG_Player_Collab
				}
				clr_country_flag = ENG_Player_Collab
				set_country_flag = ENG_Player_HMMLR
			}
			else_if = {
				limit = {
					NOT = {
						OR = {
							has_country_flag = ENG_Player_Collab
							has_country_flag = ENG_Player_HMMLR
						}
					}
				}
				set_country_flag = ENG_Player_HMMLR
			}
			ENG_GoogleMaps_Math = yes
		}
	}
	ENG_DEC_HMMLROpposition = {
		icon = GFX_decision_generic_prepare_civil_war
		available = {
			hidden_trigger = {
				always = no
			}
		}
		activation = {
			hidden_trigger = {
			AND = {
				has_country_flag = ENG_Player_Collab
				has_country_flag = ENG_GoogleMaps_ActiveDecisions
			} }
		}
		visible = {
			hidden_trigger = {
			AND = {
				has_country_flag = ENG_Player_Collab
				has_country_flag = ENG_GoogleMaps_ActiveDecisions
			} }
		}

		fixed_random_seed = no
		cancel_if_not_visible = yes
		is_good = yes
		days_mission_timeout = 15
		days_remove = 5
		days_re_enable = 10
		timeout_effect = {
			hidden_effect = {
				every_owned_state = {
					set_variable = { THIS.ENG_StateStats_Opposition = random }
					if = {
						limit = {
							check_variable = { THIS.ENG_StateStats_Opposition < 0.60 }
							date < 1963.2.2
						}
						add_to_variable = { THIS.ENG_StateStats_RawHmmlrSupportTrend = 0.5 }
						hidden_effect = {
							ENG_GoogleMaps_Math = yes
						}
					}
					else_if = {
						limit = { check_variable = { THIS.ENG_StateStats_Opposition > 0.80 } }
						add_to_variable = { THIS.ENG_StateStats_Stability = -5 }
						ENG_SE_ClampStateStability = yes
					}
					else = {
						add_to_variable = { THIS.ENG_StateStats_HmmlrSupport = 2.5 }
						ENG_SE_ClampHmmlrSupport = yes
					}
				}
			}
		}

		ai_will_do = {
			factor = 10
		}
	}
