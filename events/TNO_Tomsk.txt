country_event = { #house elections
	id = ERS.0
	title = ERS.0.t
	desc = ERS.0.desc
	immediate = {
		log = "[GetDateText]: [Root.GetName]: event ERS.0"
		hidden_effect = {
			ERS = {
				ERS_save_current_values = yes
				if = {
					limit = {
						OR = {
							has_country_flag = Decembrist_Constitution
							has_country_flag = Bastillard_Constitution
						}
					}
					ERS_Dec_and_Bas_get_upper_house = yes
				}
				if = {
					limit = {
						NOT = {
							has_country_flag = Modernist_Constitution
						}
					}
					ERS_get_lower_house = yes
				}
				if = {
					limit = {
						has_country_flag = Modernist_Constitution
					}
					ERS_Mod_get_ministers = yes
				}
				ERS_calculate_new_values = yes
				if = {
					limit = {
						NOT = { has_government = social_conservative }
					}
					ERS_Decrease_Dec_upper_house_support_low = yes
					ERS_Decrease_Dec_lower_house_support_low = yes
				}
				else_if = {
					limit = {
						NOT = { has_government = social_liberal }
					}
					ERS_Decrease_Mod_upper_house_support_low = yes
					ERS_Decrease_Mod_lower_house_support_low = yes
				}
				else_if = {
					limit = {
						NOT = { has_government = authoritarian_democrat }
					}
					ERS_Decrease_Bas_upper_house_support_low = yes
					ERS_Decrease_Bas_lower_house_support_low = yes
				}
				else_if = {
					limit = {
						NOT = { has_government = social_democrat }
					}
					ERS_Decrease_Hum_upper_house_support_low = yes
					ERS_Decrease_Hum_lower_house_support_low = yes
				}
			}
		}
	}
	
	is_triggered_only = yes

	option = {
		name = ERS.0.a
		hidden_effect = {
			ERS = {
				country_event = { id = ERS.0 days = 365 }
				ERS_clear_all_values = yes
			}
		}
	}
}
