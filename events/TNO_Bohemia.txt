country_event = { # Fight for prague hidden calculator
	id = CZE.38
	title = CZE.38.t
	desc = CZE.38.d
	picture = GFX_report_event_german_Army

	hidden = yes
	is_triggered_only = yes

	immediate = { log = "[GetDateText]: [Root.GetName]: event CZE.38" }

	option = {
		name = CZE.38.a
		set_temp_variable = { SS_prague_control = CZE_ss_control }
		subtract_from_temp_variable = { SS_prague_control = 9 }
		divide_temp_variable = { SS_prague_control = 2 }
		if = {
			limit = { has_completed_focus = CZE_claim_the_throne }
			add_to_temp_variable = { SS_prague_control = 1 }
		}
		if = {
			limit = { has_completed_focus = CZE_alone_in_the_dark }
			add_to_temp_variable = { SS_prague_control = 1 }
		}
		if = {
			limit = { has_completed_focus = CZE_blood_on_the_horizon }
			add_to_temp_variable = { SS_prague_control = 1 }
		}
		set_temp_variable = { garrison_prague_control = CZE_garrison_control }
		subtract_from_temp_variable = { garrison_prague_control = 10 }
		divide_temp_variable = { garrison_prague_control = 2 }
		add_to_temp_variable = { garrison_prague_control = CZE_guard_competence }
		if = {
			limit = { check_variable = { SS_prague_control > garrison_prague_control } }
			country_event = { id = CZE.37 }
		}
		else = {
			country_event = { id = CZE.36 }
		}
	}
}
