country_event = { 
	id = socintern.14
	immediate = {log = "[GetDateText]: [Root.GetName]: event socintern.14"}
	title = socintern.14.t
	desc = socintern.14.d
	picture = GFX_report_event_komi_government

	is_triggered_only = yes

	option = {
		name = socintern.14.a
		trigger = {
			NOT = {
				has_country_flag = SOCINTERN_LEADER
			}
		}
	}
	option = {
		name = socintern.14.b #Approve
		trigger = {
			has_country_flag = SOCINTERN_LEADER
		}
		set_global_flag = SOCINTERN_VOTE_IN_PROGRESS
		set_global_flag = SOCINTERN_VOTE_NON_AGGRESSION_PACT
		every_country = {
			limit = {
				has_idea = NIK_socintern_member
			}
			country_event = {
				id = socintern.18
				days = 3
			}
		}
	}
	option = {
		name = socintern.14.c #Veto
		trigger = {
			has_country_flag = SOCINTERN_LEADER
		}
		every_other_country = {
			limit = {
				has_idea = NIK_socintern_member
			}
			country_event = {
				id = socintern.15
				days = 3
			}
		}
		set_global_flag = SOCINTERN_VOTE_IN_PROGRESS
		set_global_flag = SOCINTERN_VOTE_NON_AGGRESSION_PACT_VETO
	}
}
